git config --global user.name WojtekSasiela <br>
git config --global user.email wojtek.sasiela@gmail.com <br>
git config --global credential.helper wincred <br>

git config --global alias.a '!git config --list | grep 'alias\\.' | sed 's/alias\\.\\([^=]*\\)=\\(.*\\)/\\1\\t=> \\2/' | sort' <br>
git config --global alias.u '!git fetch origin && git rebase origin/$1' <br>
git config --global alias.co = checkout <br>
git config --global alias.cb '!git checkout -b' <br>
git config --global alias.re = 'reset --hard' <br>
git config --global alias.res = 'reset HEAD~1' <br>
git config --global alias.r '!git remote -v' <br>
git config --global alias.b 'branch -v' <br>
git config --global alias.s 'status -sb' <br>
git config --global alias.p 'push -u origin' <br>
git config --global alias.bm 'branch -m' <br>
git config --global alias.bd 'branch -D' <br>
git config --global alias.lg 'log --oneline' <br>
git config --global alias.c '!git commit -am' <br>
git config --global alias.i '!git cb dev # && git c "Initialize project"' <br>
git config --global alias.save !git add -A && !git commit -m 'SAVEPOINT' <br>
git config --global alias.undo 'reset HEAD~1 --mixed' <br>
git config --global alias.vim 'config --global -e' <br>
git config --global alias.reset '!git reset --hard' <br>
git config --global alias.st '!git stash' <br>
git config --global alias.sta '!git stash apply' <br>
git config --global alias.fo '!git fetch origi'n <br>
git config --global alias.log 'log --pretty=format:"%h %ad | %s%d [%an]" --graph --date=short' <br>
